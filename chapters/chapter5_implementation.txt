5.1 IMPLEMENTATION OVERVIEW
-----------------------------

The system is implemented using Python as the core programming language.
The implementation follows a modular approach where each functionality is
developed as a separate module and integrated through the Flask web
framework.

The development process followed these steps:
  1. Setting up the Python virtual environment and installing dependencies
  2. Implementing the face detection and recognition engine
  3. Building the face registration module (live camera + upload fallback)
  4. Implementing the two-step training workflow (capture then train)
  5. Developing the attendance logging system
  6. Creating the Flask web application and routes
  7. Designing the HTML/CSS/JS frontend templates
  8. Adding interactive alerts with lightbox modal
  9. Integrating all modules and testing
  10. Patching face_recognition_models for Python 3.14 compatibility

  D) FACE RECOGNITION ENGINE (face_rec_engine.py) - KEY CODE:
  -------------------------------------------------------------

  import face_recognition
  import cv2
  import numpy as np
  import pickle
  import os
  from datetime import datetime

  class FaceRecognitionEngine:
      def __init__(self, encodings_path, tolerance=0.6):
          self.encodings_path = encodings_path
          self.tolerance = tolerance
          self.known_encodings = []
          self.known_names = []
          self.load_encodings()

      def load_encodings(self):
          """Load saved face encodings from file."""
          if os.path.exists(self.encodings_path):
              with open(self.encodings_path, 'rb') as f:
                  data = pickle.load(f)
              self.known_encodings = data['encodings']
              self.known_names = data['names']

      def register_face(self, image_path, name):
          """Register a new face into the system."""
          image = face_recognition.load_image_file(image_path)
          encodings = face_recognition.face_encodings(image)

          if len(encodings) > 0:
              self.known_encodings.append(encodings[0])
              self.known_names.append(name)
              self.save_encodings()
              return True
          return False

      def save_encodings(self):
          """Save face encodings to pickle file."""
          data = {
              'encodings': self.known_encodings,
              'names': self.known_names
          }
          with open(self.encodings_path, 'wb') as f:
              pickle.dump(data, f)

      def recognize_faces(self, frame):
          """Detect and recognize faces in a video frame."""
          # Resize frame for faster processing
          small_frame = cv2.resize(frame, (0, 0), fx=0.25, fy=0.25)
          rgb_frame = cv2.cvtColor(small_frame, cv2.COLOR_BGR2RGB)

          # Detect face locations and encodings
          face_locations = face_recognition.face_locations(rgb_frame)
          face_encodings = face_recognition.face_encodings(
              rgb_frame, face_locations
          )

          results = []
          for encoding, location in zip(face_encodings,
                                         face_locations):
              # Compare with known faces
              matches = face_recognition.compare_faces(
                  self.known_encodings, encoding,
                  tolerance=self.tolerance
              )
              name = "Unknown"

              if True in matches:
                  # Find best match
                  face_distances = face_recognition.face_distance(
                      self.known_encodings, encoding
                  )
                  best_match_idx = np.argmin(face_distances)
                  if matches[best_match_idx]:
                      name = self.known_names[best_match_idx]

              # Scale back face location
              top, right, bottom, left = location
              top *= 4; right *= 4; bottom *= 4; left *= 4

              results.append({
                  'name': name,
                  'location': (top, right, bottom, left),
                  'confidence': 1 - min(face_distances)
                                if True in matches else 0
              })

          return results


  E) ATTENDANCE MANAGER (attendance_manager.py) - KEY CODE:
  -----------------------------------------------------------

  import sqlite3
  from datetime import datetime, date
  import pandas as pd

  class AttendanceManager:
      def __init__(self, db_path):
          self.db_path = db_path
          self.init_db()

      def init_db(self):
          """Initialize database tables."""
          conn = sqlite3.connect(self.db_path)
          cursor = conn.cursor()

          cursor.execute('''
              CREATE TABLE IF NOT EXISTS users (
                  user_id INTEGER PRIMARY KEY AUTOINCREMENT,
                  name TEXT NOT NULL,
                  email TEXT,
                  photo_path TEXT,
                  registered_date TIMESTAMP DEFAULT
                      CURRENT_TIMESTAMP
              )
          ''')

          cursor.execute('''
              CREATE TABLE IF NOT EXISTS attendance (
                  record_id INTEGER PRIMARY KEY AUTOINCREMENT,
                  user_id INTEGER,
                  name TEXT NOT NULL,
                  date TEXT NOT NULL,
                  check_in_time TEXT NOT NULL,
                  status TEXT DEFAULT 'Present',
                  FOREIGN KEY (user_id) REFERENCES users(user_id)
              )
          ''')

          cursor.execute('''
              CREATE TABLE IF NOT EXISTS unknown_faces (
                  log_id INTEGER PRIMARY KEY AUTOINCREMENT,
                  image_path TEXT,
                  timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                  alert_status TEXT DEFAULT 'Unreviewed'
              )
          ''')

          conn.commit()
          conn.close()

      def mark_attendance(self, name, user_id=None):
          """Mark attendance for a recognized person."""
          today = date.today().strftime('%Y-%m-%d')
          now = datetime.now().strftime('%H:%M:%S')

          conn = sqlite3.connect(self.db_path)
          cursor = conn.cursor()

          # Check if already marked today
          cursor.execute(
              'SELECT * FROM attendance WHERE name=? AND date=?',
              (name, today)
          )
          if cursor.fetchone() is None:
              cursor.execute(
                  '''INSERT INTO attendance
                     (user_id, name, date, check_in_time)
                     VALUES (?, ?, ?, ?)''',
                  (user_id, name, today, now)
              )
              conn.commit()
              conn.close()
              return True
          conn.close()
          return False  # Already marked

      def get_attendance(self, date_filter=None):
          """Get attendance records, optionally filtered by date."""
          conn = sqlite3.connect(self.db_path)
          if date_filter:
              df = pd.read_sql_query(
                  'SELECT * FROM attendance WHERE date=?',
                  conn, params=[date_filter]
              )
          else:
              df = pd.read_sql_query(
                  'SELECT * FROM attendance ORDER BY date DESC',
                  conn
              )
          conn.close()
          return df

      def export_to_csv(self, date_filter=None, filepath=None):
          """Export attendance to CSV file."""
          df = self.get_attendance(date_filter)
          if filepath is None:
              filepath = f'reports/attendance_{date_filter or "all"}.csv'
          df.to_csv(filepath, index=False)
          return filepath

  DATABASE NAME: database.db (SQLite)

  TABLE 1: users
  +------------------+-----------+---------------------------------------+
  | Column Name      | Data Type | Constraints / Notes                   |
  +------------------+-----------+---------------------------------------+
  | user_id          | INTEGER   | PRIMARY KEY, AUTOINCREMENT            |
  | name             | TEXT      | NOT NULL                              |
  | email            | TEXT      |                                       |
  | photo_path       | TEXT      | Relative path to first preview photo  |
  | photo_dir        | TEXT      | Relative path to user's photo folder  |
  | trained          | INTEGER   | DEFAULT 0 (0=untrained, 1=trained)    |
  | registered_date  | TIMESTAMP | DEFAULT CURRENT_TIMESTAMP             |
  +------------------+-----------+---------------------------------------+

  NOTE: The 'trained' column is the key addition enabling the two-step
  registration workflow. A user with trained=0 has photos saved but their
  face encoding has NOT been added to the recognition model. Only after
  the admin clicks "Train Model" are their encodings computed and the
  column set to 1, activating recognition for that user.

  TABLE 2: attendance

5.4 USER INTERFACE DESIGN
---------------------------

  The user interface is built using HTML5, CSS3, Bootstrap 5, and
  Jinja2 templates. The key pages are:

  1. LOGIN PAGE:
     - Clean, centered login form
     - Username and password fields
     - "Login" button with form validation
     - Error message display for invalid credentials

  2. DASHBOARD (Home Page):
     - Sidebar navigation with system name and menu links
     - Summary cards: Total registered users, Today's attendance,
       Unknown face alert count
     - Quick action buttons (Register, View Attendance, etc.)
     - Live camera feed preview

  3. FACE REGISTRATION PAGE (Two-Tab Design):

     TAB 1 — Live Camera (primary workflow):
       - Name and email input fields
       - "Open Camera" button: requests browser camera permission via
         the browser-native getUserMedia API (works on localhost/HTTPS;
         blocked on plain HTTP + IP address by browser security policy)
       - Live <video> element showing real-time camera feed
       - "Start Capture" button: auto-captures 50 frames at 350ms
         intervals using a hidden <canvas> element that draws from the
         live video (no server-side camera conflict)
       - Real-time progress bar (0-100%, clamped to avoid >100%)
       - Frame counter badge overlaid on the video
       - White camera-flash effect every captured frame
       - Thumbnail strip showing every 5th captured frame
       - On completion: frames uploaded as base64 JSON to /register_camera
         which saves them to static/registered_faces/<name>/ and adds the
         user as UNTRAINED (trained=0) in the database

     TAB 2 — Upload Photo (fallback):
       - Single photo file upload (JPG/PNG)
       - Image preview before submitting
       - Submits to /register route for immediate single-photo encoding

     TRAIN MODEL PANEL (below registration form, always visible):
       - Info banner explaining the two-step workflow
       - Counts and lists all untrained users (auto-loaded on page open
         via /untrained_users JSON endpoint)
       - Table shows: photo thumbnail, name, email, registration date
       - Badge showing number of untrained users (yellow) or "0 untrained"
         (green when all are trained)
       - "Train Model" button: POSTs to /train_model which reads all
         saved photos, extracts and averages face encodings, saves the
         updated model, and marks users as trained=1
       - Animated progress bar and spinner shown during training
       - Success/failure message with per-user training results

  4. ATTENDANCE PAGE:
     - Date filter dropdown/picker
     - Table showing: Name, Date, Check-in Time, Status
     - Pagination for large records
     - "Export to CSV" button
     - Search functionality

  5. ALERTS PAGE:
     - Filter buttons: All / Unreviewed / Reviewed
     - Photo card grid with hover zoom effect
     - Status badge on each card (red "New" or grey "Reviewed")
     - Click any photo → opens a full dark lightbox modal with:
         * Large image display (max 500px height, object-fit: contain)
         * Timestamp and status badge
         * Download button for saving the unknown face image
         * Close button
     - "Mark Reviewed" button on each card (for unreviewed alerts)

  6. REPORTS PAGE:
     - Date range selector
     - Generate report button
     - Summary statistics display
     - Download options (CSV)
